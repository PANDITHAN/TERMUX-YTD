# INSTALLATION TOOL
YouTube vedio download tool using termux

$ pkg update

$ pkg upgrade

$ pkg install python

$ pkg intall python2

$ pkg install git -y

$ git clone https://github.com/PANDITHAN/TERMUX-YTD/

$ ls

$ cd TERMUX-YTD

$ ls

$ chmod +x *

$ bash install.sh
